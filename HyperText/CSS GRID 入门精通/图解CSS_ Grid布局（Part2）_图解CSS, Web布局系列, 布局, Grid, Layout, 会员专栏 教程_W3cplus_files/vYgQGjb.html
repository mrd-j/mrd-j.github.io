let inputRadios = document.querySelectorAll('input[type="radio"]');
let inputRanges = document.querySelectorAll('input[type="range"]');

let gridContainer = document.querySelector(".grid__container");

inputRadios.forEach((input, index) => {
  input.addEventListener("click", (etv) => {
    gridContainer.style.setProperty("--display", etv.target.id);
  });
});

inputRanges.forEach((input, index) => {
  input.addEventListener("click", (etv) => {
    gridContainer.style.setProperty(`--${etv.target.id}`, etv.target.value);
    document.getElementById(`${etv.target.id}-output`).textContent =
      etv.target.value;
  });
});
