inputs.forEach((input, index) => {
  input.addEventListener("click", (etv) => {
    if (etv.target.id === "area__before") {
      container.style.gridTemplateAreas = `"header header" "sidebar main" "footer footer"`;
    } else if (etv.target.id === "area__after") {
      container.style.gridTemplateAreas = `"header header" "sidebar main" "sidebar footer"`;

      console.log("grid-template-areas");
