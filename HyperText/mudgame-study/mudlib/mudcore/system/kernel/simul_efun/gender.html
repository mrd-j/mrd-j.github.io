string pronoun(int pronoun_number, mixed arg)
{
    if( objectp(arg) )
        arg = arg->query("gender");

    switch(pronoun_number)
    {
        case 1:
            return "我";

        case 2:
            return "你";

        case 3:
            switch( arg )
            {
                case MALE: return "他";
                case FEMALE: return "她";
                default:
                    return "它";
            }
        default:
            error("错误的人称代词。\n");
    }
}
