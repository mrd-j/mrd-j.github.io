mapping expand_keys(mapping mp)
{
    mapping tmp = ([]);
    mixed key, val;

    foreach (key, val in mp)
    {
        if (arrayp(key))
        {
            string elem;

            foreach (elem in key)
                tmp[elem] = val;
        }
        else
            tmp[key] = val;
    }
    return tmp;
}
