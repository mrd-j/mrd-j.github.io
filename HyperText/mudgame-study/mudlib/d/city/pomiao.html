int valid_leave(object me, string dir)
{
    object guard;

    if (dir != "in" || !objectp(guard = present("lu youjiao", this_object())) && !objectp(guard = present("peng youjing", this_object())))
        return ::valid_leave(me, dir);

    return guard->permit_pass(me, dir);
}
