int valid_leave(object me, string dir)
{
    if (me->query("gender") != "男性" && dir == "west")
        return notify_fail("男女有别，请到对面住！\n");

    if (me->query("gender") != "女性" && dir == "east")
        return notify_fail("男女有别，请到对面住！\n");

    return ::valid_leave(me, dir);
}
