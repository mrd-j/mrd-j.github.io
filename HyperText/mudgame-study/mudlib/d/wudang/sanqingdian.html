int valid_leave(object me, string dir)
{
        object guarder;

        if (dir != "west" && dir != "south")
                return 1;

        if (! objectp(guarder = present("song yuanqiao", environment(me))))
                return 1;

        return guarder->permit_pass(me, dir);
}
