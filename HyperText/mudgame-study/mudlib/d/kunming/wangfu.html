int valid_leave(object me, string dir)
{
        if (dir == "north" &&
           objectp(present("qin bing", environment(me))))
                return notify_fail(CYN "亲兵拦住你说道：王府大院，闲杂人"
                                   "等一律回避。\n" NOR);                
        return ::valid_leave(me, dir);
}
