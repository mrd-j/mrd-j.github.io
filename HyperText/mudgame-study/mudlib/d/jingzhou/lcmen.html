int valid_leave(object me, string dir)
{
        if (dir == "east" &&
           objectp(present("guan bing", environment(me))))
                return notify_fail(CYN "官兵一把拦住你，骂道：粮仓重地，岂是你们这些"
                                   "泥腿子随便来得的？\n" NOR);
        return ::valid_leave(me, dir);
}
