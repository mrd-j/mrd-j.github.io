{

        set("combat_exp", 80000);
        set("shen_type", -1);

        set_skill("pixie-sword", 80);
        set_skill("unarmed", 80);
        set_skill("dodge", 80);
        set_skill("parry", 80);
        map_skill("parry", "pixie-sword");
        map_skill("sword", "pixie-sword");

        carry_object("/clone/weapon/changjian")->wield();
