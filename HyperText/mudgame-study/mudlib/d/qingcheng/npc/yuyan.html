{

    set_skill("pixie-sword", 40);
    set_skill("unarmed", 40);
    set_skill("dodge", 40);
    set_skill("parry", 40);
    map_skill("parry", "pixie-sword");

    carry_object("/clone/weapon/changjian")->wield();
