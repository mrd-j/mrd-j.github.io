{

        if (clonep())
                set_default_object(__FILE__);
        else {

                set("value", 25000);
                set("material", "steel");

                set("unwield_msg", HIY "$N" HIY "微微一笑，将手中的金笛插回腰间。\n" NOR);

        init_sword(30);
        setup();
}
