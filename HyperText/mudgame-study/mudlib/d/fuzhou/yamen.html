int valid_leave(object me, string dir)
{
        object ob;

    if (objectp(ob = present("ge shiha", environment(me))) &&
            dir != "east" && living(ob))
        return notify_fail("戈什哈喝道：“施大人已去校场阅兵，你来何干？”\n");

    return ::valid_leave(me, dir);
}
