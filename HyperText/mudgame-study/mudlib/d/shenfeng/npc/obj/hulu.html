{

        set_weight(700);
        if (clonep())
                set_default_object(__FILE__);
        else {
                set("long", "一个用来装酒的大葫芦，外皮金黄，里面能装好几斤酒。\n");
                set("unit", "个");
                set("value", 100);
                set("max_liquid", 15);
        }
        set("liquid", ([
                "type": "alcohol",
                "name": "米酒",
                "remaining": 15,
                "drunk_apply": 3,
        ]));
}
