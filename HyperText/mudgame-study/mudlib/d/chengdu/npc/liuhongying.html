        set_skill("strike", 180);

        set_skill("dagger", 180);

        set_temp("apply/defense", 80);
        set_temp("apply/damage", 20);

int recognize_apprentice(object ob)
{
        if (!(int)ob->query_temp("marks/jinniu_paied") == 1) return 0;
        return 1;
}
