int valid_leave(object me, string dir)
{
        if (objectp(present("riyue dizi", environment(me)))
           && dir == "west"
           && me->query("family/family_name") != "日月神教")
                return notify_fail(CYN "日月弟子将你拦住，说道：这里乃神"
                                   "教重地，他人不得入内。\n" NOR);

        return ::valid_leave(me, dir);
}
