int valid_leave(object me, string dir)
{
        object guarder;

        if (dir != "north")
                return 1;

        if (objectp(guarder = present("mei jian", environment(me))))
                return guarder->permit_pass(me, dir);

        if (objectp(guarder = present("lan jian", environment(me))))
                return guarder->permit_pass(me, dir);

        return 1;
}
