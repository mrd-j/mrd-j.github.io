int valid_leave(object me, string dir)
{
    object guarder;

    if (!objectp(guarder = present("chuchen zi", this_object())))
        return 1;
    // todo ???
    if (dir == "in" || (dir == "in" && !userp(me)))
        return guarder->permit_pass(me, dir);

    return 1;
}
