int valid_leave(object me, string dir)
{
    object ob;
    if (! objectp(ob = present("lu tianshu", this_object())))
    {
        ob = new(__DIR__"npc/lu");
        ob->move(this_object());
    }

    ob->check_leave(me, dir);
    return ::valid_leave(me, dir);
}
