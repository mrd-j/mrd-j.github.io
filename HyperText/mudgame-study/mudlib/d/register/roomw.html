int valid_leave(object me, string dir)
{
    object ob;
    if (!objectp(ob = present("hua tiegan", this_object())))
    {
        ob = new (__DIR__ "npc/hua");
        ob->move(this_object());
    }

    ob->check_leave(me, dir);
    return ::valid_leave(me, dir);
}
