{

    if (clonep())

    else {

        set("unit", "块");
        set("value", 30000);
                set("item_origin", 1);
                set("material_attrib", "steel");
                set("material_name", CYN "玄铁" NOR);
                set("can_make", "all");
                set("not_make", ({ "鞭", "战衣", "锤", "腰带" }));
                set("power_point", 70);
    }
}
