int valid_leave(object me, string dir)
{
    string dest;

    if (interactive(me) && dir == "north")
    {
        dest = __DIR__"gebitan";
        dest += "" + (string) (random(5) + 1);
        me->move(dest);
        return -1;
    }
    return ::valid_leave(me, dir);
}
