{

    set_weight(30000);
    set_max_encumbrance(5000);
    if( clonep() )
        set_default_object(__FILE__);
    else {
        set("unit", "具" );

        set("no_get", "人都死了，何必还烦动她的尸骨？\n");
    }
    setup();
}
