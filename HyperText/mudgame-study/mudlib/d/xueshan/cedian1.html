int valid_leave(object me, string dir)
{
        mapping myfam;
        myfam = (mapping)me->query("family");

        if ((! myfam || myfam["family_name"] != "大轮寺")
           && dir=="northup"
           && objectp(present ("jiamu huofo", environment(me))))
                return notify_fail(CYN "嘉木活佛说道：尔非大轮寺弟"
                                   "子，不能入内。\n" NOR);

        return ::valid_leave(me, dir);
}
