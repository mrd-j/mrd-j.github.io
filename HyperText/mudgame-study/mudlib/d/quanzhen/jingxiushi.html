int valid_leave(object me, string dir)
{
        mapping fam;
        object ob;

        if (! (fam = me->query("family"))
           || fam["family_name"] != "全真教")
        {
                if ((ob = present("liu chuxuan", environment(me)))
                   && living(ob)
                   && dir == "east")
                        return notify_fail(CYN "刘处玄伸手拦住你道：后面是本"
                                           "教祖师清修之地，请止步。\n" NOR);
        }
        return ::valid_leave(me, dir);
}
