void create()
{
        set_name("木琴", ({ "muqin" }));
        set_weight(600);

    if (clonep())
        set_default_object(__FILE__);
    else {
                set("unit", "台");

        set("value", 50);
        set("material", "wood");
    }
    setup();
}

void init()
{
        add_action("play_qin", "play");
}
