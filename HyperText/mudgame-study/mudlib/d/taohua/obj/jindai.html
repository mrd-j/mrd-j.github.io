void create()
{
        set_name( "束发金带" , ({ "jindai" }) );
        set_weight(50);
        if( clonep() )
                set_default_object(__FILE__);

                set("material", "silk");
                set("armor_prop/armor", 1);
        set("wear_msg", "$N将$n拿出束在头发上。\n");
        }
        setup();
}
