inherit FORCE;

int valid_enable(string usage) 
{ 
        return usage == "force"; 
}

int valid_force(string force)
{
        return force == "yunv-xinjing" ||
               force == "xuanmen-neigong" ||
               force == "xiantian-gong" ||
               force == "quanzhen-xinfa";
}

{
            return ::valid_learn(me);
}

}
