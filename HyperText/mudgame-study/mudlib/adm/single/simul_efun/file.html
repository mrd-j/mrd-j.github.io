void cat(string file)
{
    if (previous_object())
        seteuid(geteuid(previous_object()));
    else
        seteuid(ROOT_UID);
    write(read_file(file));
}

varargs void log_file(string file, string text, int flag)
{
    seteuid(ROOT_UID);
    CORE_SIMUL_EFUN_OB->log_file(file, text, flag);
}
