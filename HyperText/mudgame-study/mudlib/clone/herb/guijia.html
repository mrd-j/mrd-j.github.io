#include "herb.h"

void create()
{
        set_name(NOR + CYN "龟甲" NOR, ({ "gui jia", "gui", "jia", "herb_guijia" }));
        if (clonep())
                set_default_object(__FILE__);

                set("base_unit", "块");

        }
        setup();
}
