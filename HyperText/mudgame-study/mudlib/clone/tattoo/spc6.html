        if (clonep())

                set("material", "paper");
                set("tattoo_long", "大慈大悲观世音菩萨，坐八宝莲花，千支手姿态各异");

                set("tattoo_type", ({ "悟性" }));
                set("tattoo_int", 4);
                set("tattoo_special", "clever");
                set("tattoo_spcname", "天赋聪颖");

}
