{

        else {

                set("base_value", 1);
                set("material", "iron");
                set("long", HIW "铁莲子是一种以量取胜的暗器，沉颠颠的。\n" NOR);
                set("wield_msg", HIC "$N" HIC "迅速地从暗器袋中掏出一"

                                   "放回暗器袋中。\n"NOR);
