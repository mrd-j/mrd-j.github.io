{
        set_name(HIW "獨角獸之角" NOR, ({ "unicorn horn", "unicorn", "horn" }));
        set_weight(1000);
        if (clonep())
                set_default_object(__FILE__);

                set("unit", "根");
                set("value", 700000);
        }
        setup();
}
